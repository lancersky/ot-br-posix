342 template <typename T> otbrError DBusMessageExtract(DBusMessageIter *aIter, std::vector<T> &aValue)
343 {
344  otbrError error = OTBR_ERROR_NONE;
345  DBusMessageIter subIter;
346 
347  VerifyOrExit(dbus_message_iter_get_arg_type(aIter) == DBUS_TYPE_ARRAY, error = OTBR_ERROR_DBUS);
348  dbus_message_iter_recurse(aIter, &subIter);
349 
350  aValue.clear();
351  while (dbus_message_iter_get_arg_type(&subIter) != DBUS_TYPE_INVALID)
352  {
353  T val;
354  SuccessOrExit(error = DBusMessageExtract(&subIter, val));
355  aValue.push_back(val);
356  }
357  dbus_message_iter_next(aIter);
358 
359 exit:
360  return error;
361 }
